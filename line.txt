const sendLineMessage = (userId, message)=>  {  //引数で指定されたユーザに対して、引数で指定されたメッセージを送信する
  const url = "https://api.line.me/v2/bot/message/push";  //指定されたユーザにメッセージを送信するためのURL
  
  const headers = {
    "Authorization": 'Bearer ' + ACCESS_TOKEN,
    "Accept": "application/json",
    "Content-type": "application/json"
  };  

  const payload = {
    "to" : userId,
    "messages":[{
      "type": "text",
      "text":  message 
    }]
  };

  const options = {
    "method": "post",
    "payload": JSON.stringify(payload),
    "headers": headers
  };

  UrlFetchApp.fetch(url, options);  //指定されたユーザにメッセージを送信
}
